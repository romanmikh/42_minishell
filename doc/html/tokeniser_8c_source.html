66void handle_phrase(char **str, t_token **tokens)
67{
68 char *start;
69 char quote_flag;
70 char quote_type;
71
72 start = *str;
73 quote_flag = 0;
74 quote_type = 0;
75 while (**str)
76 {
77 if (!quote_flag && (**str == '\'' || **str == '\"'))
78 {
79 quote_flag = 1;
80 quote_type = **str;
81 }
82 else if (quote_flag && **str == quote_type)
83 quote_flag = 0;
84 if (!quote_flag && ft_strchr(" \t\n\r\v\f<|>", **str) != NULL)
85 break ;
86 (*str)++;
87 }
88 append_phrase_if_valid(&start, str, tokens);
89}

107{
108 t_token *tokens;
109
